263  void handle_message(const network_message::game_message* msg) {
264  // threads block until they acquire lock
265  std::lock_guard<std::mutex> lock(game_mutex);
266  ssize_t result = 1;
267 
268  if (msg->type() == type::ERROR) {

271  }
272 
273  switch (msg->signal()) {
274 
275  // handle game creation and player turn order
276  case signal::GAME_CREATED: {
277  if (!msg->has_payload()) {
278  break;
279  }
280  fill_player_board(c_to_p(msg->from()), msg->payload());
281 
282  if (c_to_p(msg->from()) == 0) {
283  send_game_message(p_to_c(0), signal::GAME_PLAYER_ONE);
284  } else {
285  send_game_message(p_to_c(1), signal::GAME_PLAYER_TWO);
286  }
287  }
288  break;
289

291  if (!msg->has_payload()) {
292  break;
293  }
294  unsigned char index = *msg->payload();
295  signal sig;
296  const unsigned char square = get_square(c_to_p(msg->from()),
297  index);

302  sig = signal::GAME_HIT;
303  }
304  const char opp = 1 - c_to_p(msg->from());
305  send_game_message(p_to_c(msg->from()), sig);
306 
307  send_game_message(p_to_c(opp), sig);
308  send_game_message(p_to_c(opp), signal::GAME_COORDINATES,

311  }
